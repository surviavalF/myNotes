子组件
```vue
<template>
	<button @click="getChild">触发子组件方法</button>
	<!-- 一：定义 ref -->
	<Child ref="childRef"></Child>
</template>
<script lang="ts" setup>
	import { ref } from 'vue';
	import Child from '../../components/child.vue';
	// 二：定义与 ref 同名变量
	const childRef = ref <any> ()
	// 三、函数
	const getChild = () => {
		// 调用子组件的方法或者变量，通过value
		childRef.value.hello("hello world！");
	}
</script>
```
